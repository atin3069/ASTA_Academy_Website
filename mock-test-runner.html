// ============ STATE ============ let testData = null; let scoringSpec = null; let currentModule = null; let currentModuleQuestions = []; let currentQuestionIdx = 0; let answers = {}; // { questionId: chosenAnswer } let m1Result = null; let m2Variant = null; let m2Result = null; let timerInterval = null; let timeRemaining = 0; let testStartTime = null; let m1TimeUsed = 0; let m2TimeUsed = 0; const WEIGHTS = { Easy: 1.0, Medium: 1.5, Hard: 2.0 }; const ROUTING_THRESHOLD = 18.0; // ============ INIT ============ async function init() { try { const [td, sd] = await Promise.all([ fetch('mock-test-1.json').then(r => r.json()), fetch('scoring-spec.json').then(r => r.json()) ]); testData = td; scoringSpec = sd; } catch (e) { alert('Could not load test data. Make sure mock-test-1.json and scoring-spec.json are in the same folder.'); } } init(); // ============ SCREEN MGMT ============ function showScreen(id) { document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); document.getElementById(id).classList.add('active'); document.getElementById('test-header').style.display = (id === 'screen-question') ? 'flex' : 'none'; } // ============ MODULE START ============ function startModule(num) { if (num === 1) { currentModule = 1; currentModuleQuestions = testData.module_1.questions; document.getElementById('header-module').textContent = 'Module 1'; } else { currentModule = 2; const variantData = m2Variant === 'hard' ? testData.hard_module_2 : testData.easy_module_2; currentModuleQuestions = variantData.questions; document.getElementById('header-module').textContent = 'Module 2 (' + (m2Variant === 'hard' ? 'Hard' : 'Easy') + ')'; } currentQuestionIdx = 0; testStartTime = Date.now(); startTimer(35 * 60); document.getElementById('total-q-num').textContent = currentModuleQuestions.length; showScreen('screen-question'); renderQuestion(); } // ============ TIMER ============ function startTimer(seconds) { timeRemaining = seconds; updateTimerDisplay(); if (timerInterval) clearInterval(timerInterval); timerInterval = setInterval(() => { timeRemaining--; updateTimerDisplay(); if (timeRemaining <= 0) { clearInterval(timerInterval); submitModule(); } }, 1000); } function updateTimerDisplay() { const m = Math.floor(timeRemaining / 60); const s = timeRemaining % 60; const display = `${m}:${s.toString().padStart(2, '0')}`; const timer = document.getElementById('timer'); timer.textContent = display; timer.classList.toggle('warning', timeRemaining < 300); } // ============ QUESTION DISPLAY ============ function renderQuestion() { const q = currentModuleQuestions[currentQuestionIdx]; document.getElementById('current-q-num').textContent = currentQuestionIdx + 1; document.getElementById('domain-label').textContent = q.domain; document.getElementById('q-id').textContent = q.id; document.getElementById('q-difficulty').textContent = q.difficulty; document.getElementById('q-difficulty').className = 'difficulty ' + q.difficulty; document.getElementById('q-text').textContent = q.question; document.getElementById('progress-fill').style.width = ((currentQuestionIdx + 1) / currentModuleQuestions.length * 100) + '%'; const answerArea = document.getElementById('answer-area'); answerArea.innerHTML = ''; if (q.type === 'MC' || q.choices) { const choicesDiv = document.createElement('div'); choicesDiv.className = 'choices'; ['A', 'B', 'C', 'D'].forEach(letter => { if (q.choices && q.choices[letter] !== undefined) { const btn = document.createElement('button'); btn.className = 'choice-btn' + (answers[q.id] === letter ? ' selected' : ''); btn.dataset.letter = letter; btn.innerHTML = `${letter}${q.choices[letter]}`; btn.onclick = () => selectChoice(letter); choicesDiv.appendChild(btn); } }); answerArea.appendChild(choicesDiv); } else { // SPR const input = document.createElement('input'); input.type = 'text'; input.className = 'spr-input'; input.placeholder = 'Type your answer'; input.value = answers[q.id] || ''; input.oninput = (e) => { answers[q.id] = e.target.value.trim(); updateNavStatus(); }; answerArea.appendChild(input); const hint = document.createElement('p'); hint.className = 'spr-hint'; hint.textContent = 'Enter a number. No units, no commas.'; answerArea.appendChild(hint); } document.getElementById('prev-btn').disabled = currentQuestionIdx === 0; const nextBtn = document.getElementById('next-btn'); if (currentQuestionIdx === currentModuleQuestions.length - 1) { nextBtn.textContent = 'Submit Module ' + currentModule + ' →'; nextBtn.classList.add('submit'); } else { nextBtn.textContent = 'Next →'; nextBtn.classList.remove('submit'); } updateNavStatus(); } function selectChoice(letter) { const q = currentModuleQuestions[currentQuestionIdx]; answers[q.id] = letter; document.querySelectorAll('.choice-btn').forEach(b => { b.classList.toggle('selected', b.dataset.letter === letter); }); updateNavStatus(); } function updateNavStatus() { const q = currentModuleQuestions[currentQuestionIdx]; const answered = answers[q.id] !== undefined && answers[q.id] !== ''; document.getElementById('nav-status').textContent = answered ? '✓ Answered' : 'Skip or answer to continue'; } function prevQuestion() { if (currentQuestionIdx > 0) { currentQuestionIdx--; renderQuestion(); } } function nextQuestion() { if (currentQuestionIdx < currentModuleQuestions.length - 1) { currentQuestionIdx++; renderQuestion(); } else { submitModule(); } } // ============ MODULE SUBMISSION ============ function submitModule() { if (timerInterval) clearInterval(timerInterval); // BUG FIX 1: Capture any unsaved SPR input if time ran out while typing const activeSpr = document.querySelector('.spr-input'); if (activeSpr && activeSpr.value.trim() !== '') { const q = currentModuleQuestions[currentQuestionIdx]; if (q) answers[q.id] = activeSpr.value.trim(); } const elapsedSec = Math.floor((Date.now() - testStartTime) / 1000); const elapsedMin = Math.round(elapsedSec / 60); const result = scoreModule(currentModuleQuestions); if (currentModule === 1) { m1TimeUsed = elapsedMin; m1Result = result; m2Variant = result.weighted >= ROUTING_THRESHOLD ? 'hard' : 'easy'; showRoutingScreen(); } else { m2TimeUsed = elapsedMin; m2Result = result; showScoreScreen(); } } // Smart answer matching - handles SPR equivalents like 550 = 550.0 = 550.00, 1/2 = 0.5, etc. function isAnswerCorrect(userAns, correctAns, question) { if (!userAns) return false; // For multiple choice, simple letter comparison (case-insensitive) if (question.choices) { return userAns.toLowerCase() === correctAns.toLowerCase(); } // For SPR: try numeric equivalence first const userNum = parseAnswerToNumber(userAns); const correctNum = parseAnswerToNumber(correctAns); if (userNum !== null && correctNum !== null) { // Both are numbers - compare with tolerance for floating point return Math.abs(userNum - correctNum) < 0.0001; } // Fallback: case-insensitive string match return userAns.toLowerCase() === correctAns.toLowerCase(); } // Parse "550", "550.0", "1/2", "0.5", "-3.5" into a number. Returns null if not parseable. function parseAnswerToNumber(str) { if (str === null || str === undefined) return null; const cleaned = str.toString().trim().replace(/\s+/g, ''); if (cleaned === '') return null; // Handle fractions like "1/2" or "-3/4" const fractionMatch = cleaned.match(/^(-?\d+(?:\.\d+)?)\/(-?\d+(?:\.\d+)?)$/); if (fractionMatch) { const num = parseFloat(fractionMatch[1]); const den = parseFloat(fractionMatch[2]); if (den !== 0 && !isNaN(num) && !isNaN(den)) { return num / den; } return null; } // Handle plain numbers (including decimals, negatives, leading + sign) const numMatch = cleaned.match(/^[+-]?\d+(?:\.\d+)?$/); if (numMatch) { const n = parseFloat(cleaned); return isNaN(n) ? null : n; } return null; } function scoreModule(questions) { let correct = 0; let weighted = 0; const domainStats = {}; const missedQuestions = []; questions.forEach(q => { const userAnsRaw = (answers[q.id] || '').toString().trim(); const correctAnsRaw = q.answer.toString().trim(); const isCorrect = isAnswerCorrect(userAnsRaw, correctAnsRaw, q); if (!domainStats[q.domain]) domainStats[q.domain] = { correct: 0, total: 0, missedSkills: {} }; domainStats[q.domain].total++; if (isCorrect) { correct++; weighted += WEIGHTS[q.difficulty] || 0; domainStats[q.domain].correct++; } else { missedQuestions.push(q); const skill = q.skill || q.subtopic || 'General'; if (!domainStats[q.domain].missedSkills[skill]) { domainStats[q.domain].missedSkills[skill] = 0; } domainStats[q.domain].missedSkills[skill]++; } }); return { correct, weighted, total: questions.length, domainStats, missedQuestions }; } // ============ ROUTING SCREEN ============ function showRoutingScreen() { document.getElementById('m1-correct-display').textContent = m1Result.correct; document.getElementById('m1-weighted-display').textContent = m1Result.weighted.toFixed(1); const tag = document.getElementById('module2-tag'); const explanation = document.getElementById('routing-explanation'); if (m2Variant === 'hard') { tag.textContent = 'Routed to Hard Module 2'; tag.className = 'module2-tag hard'; explanation.textContent = 'Strong Module 1 performance — you\'ve unlocked the harder Module 2 with higher score potential.'; } else { tag.textContent = 'Routed to Easy Module 2'; tag.className = 'module2-tag easy'; explanation.textContent = 'Module 2 will adapt to your current level — focus on accuracy.'; } showScreen('screen-routing'); } // ============ FINAL SCORING ============ function calculateFinalScore() { // BUG FIX 2: Handle Zero Answers const totalAnswered = Object.keys(answers).filter(k => answers[k] !== '').length; if (totalAnswered === 0) { return { estimated: 0, low: 'Incomplete', high: 'Test', isIncomplete: true }; } const table = m2Variant === 'hard' ? scoringSpec.final_scoring.hard_module_2_score_table.rows : scoringSpec.final_scoring.easy_module_2_score_table.rows; const m1 = m1Result.weighted; const m2 = m2Result.weighted; const row = table.find(r => m1 >= r.m1_score_min && m1 <= r.m1_score_max && m2 >= r.m2_score_min && m2 <= r.m2_score_max ); const estimated = row ? row.estimated_score : (m2Variant === 'hard' ? 500 : 220); return { estimated, low: estimated - 15, high: estimated + 15 }; } function showScoreScreen() { const score = calculateFinalScore(); // UI adaptation for incomplete tests if (score.isIncomplete) { document.getElementById('score-low').textContent = 'No'; document.getElementById('score-high').textContent = 'Data'; document.querySelector('.score-hero .range em').style.display = 'none'; } else { document.getElementById('score-low').textContent = score.low; document.getElementById('score-high').textContent = score.high; } // Aggregate domains const allDomains = {}; [...Object.entries(m1Result.domainStats), ...Object.entries(m2Result.domainStats)].forEach(([d, s]) => { if (!allDomains[d]) allDomains[d] = { correct: 0, total: 0, missedSkills: {} }; allDomains[d].correct += s.correct; allDomains[d].total += s.total; Object.entries(s.missedSkills).forEach(([skill, count]) => { allDomains[d].missedSkills[skill] = (allDomains[d].missedSkills[skill] || 0) + count; }); }); window._allDomains = allDomains; window._finalScore = score; // Fill hidden form fields for Formspree document.getElementById('hidden-score-range').value = score.isIncomplete ? 'Incomplete' : (score.low + '-' + score.high); document.getElementById('hidden-m1-correct').value = m1Result.correct + '/22'; document.getElementById('hidden-m1-weighted').value = m1Result.weighted.toFixed(1); document.getElementById('hidden-m2-variant').value = m2Variant === 'hard' ? 'Hard' : 'Easy'; document.getElementById('hidden-m2-correct').value = m2Result.correct + '/22'; document.getElementById('hidden-m2-weighted').value = m2Result.weighted.toFixed(1); document.getElementById('hidden-total-correct').value = (m1Result.correct + m2Result.correct) + '/44'; document.getElementById('hidden-time-used').value = (m1TimeUsed + m2TimeUsed) + ' min'; let domainSummary = ''; Object.entries(allDomains).forEach(([d, s]) => { const pct = Math.round((s.correct / s.total) * 100); domainSummary += `${d}: ${s.correct}/${s.total} (${pct}%) | `; }); document.getElementById('hidden-domain-breakdown').value = domainSummary; let missedSummary = ''; const allMissed = []; Object.entries(allDomains).forEach(([d, s]) => { Object.entries(s.missedSkills).forEach(([skill, count]) => { allMissed.push({ domain: d, skill, count }); }); }); allMissed.sort((a, b) => b.count - a.count).slice(0, 5).forEach((m, i) => { missedSummary += `${i+1}. ${m.skill} (${m.domain}) - missed ${m.count} | `; }); document.getElementById('hidden-missed-skills').value = missedSummary; // ====== Build detailed answer log for email ====== let detailed = ''; function buildModuleLog(label, questions) { let log = `\n═══════════════════════════════════════\n${label}\n═══════════════════════════════════════\n`; questions.forEach((q, i) => { const userAns = (answers[q.id] || '').toString().trim(); const correctAns = q.answer.toString().trim(); const isCorrect = isAnswerCorrect(userAns, correctAns, q); const status = userAns === '' ? '— (skipped)' : (isCorrect ? '✓ CORRECT' : '✗ INCORRECT'); log += `\nQ${i+1} [${q.id}] ${q.difficulty} · ${q.domain}`; if (q.skill) log += ` · ${q.skill}`; log += `\n"${q.question}"\n`; if (q.choices) { // Show their choice text const userChoiceText = q.choices[userAns] || '(no answer)'; const correctChoiceText = q.choices[correctAns] || correctAns; log += `Their answer: ${userAns ? userAns + ' (' + userChoiceText + ')' : '(skipped)'} ${status}\n`; if (!isCorrect) log += `Correct answer: ${correctAns} (${correctChoiceText})\n`; } else { // SPR log += `Their answer: ${userAns || '(skipped)'} ${status}\n`; if (!isCorrect) log += `Correct answer: ${correctAns}\n`; } }); return log; } detailed += buildModuleLog('MODULE 1 — DETAILED ANSWERS', testData.module_1.questions); const m2QData = m2Variant === 'hard' ? testData.hard_module_2.questions : testData.easy_module_2.questions; detailed += buildModuleLog(`MODULE 2 (${m2Variant.toUpperCase()}) — DETAILED ANSWERS`, m2QData); document.getElementById('hidden-detailed-answers').value = detailed; // ====== End detailed answer log ====== showScreen('screen-score'); } // ============ FORM SUBMIT ============ document.getElementById('report-form').addEventListener('submit', (e) => { e.preventDefault(); const form = e.target; const formData = new FormData(form); fetch(form.action, { method: 'POST', body: formData, headers: { 'Accept': 'application/json' } }).then(() => { showReport(); }).catch(() => { // Show report anyway even if formspree fails showReport(); }); }); // ============ DIAGNOSTIC REPORT ============ function showReport() { const score = window._finalScore; const domains = window._allDomains; const totalCorrect = m1Result.correct + m2Result.correct; const totalQ = 44; const totalTime = m1TimeUsed + m2TimeUsed; document.getElementById('report-meta').textContent = 'Mock Test 1 · ' + new Date().toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' }); document.getElementById('rpt-score-range').textContent = score.isIncomplete ? 'Incomplete' : (score.low + '-' + score.high); document.getElementById('rpt-total-correct').textContent = totalCorrect; document.getElementById('rpt-percent').textContent = Math.round(totalCorrect / totalQ * 100) + '%'; document.getElementById('rpt-time').textContent = totalTime; // Domain breakdown const grid = document.getElementById('domain-grid'); grid.innerHTML = ''; Object.entries(domains).forEach(([domain, stats]) => { const pct = Math.round((stats.correct / stats.total) * 100); let level, color; if (pct >= 85) { level = 'Strong'; color = '#0F6E56'; } else if (pct >= 65) { level = 'On-Track'; color = '#2563a8'; } else if (pct >= 40) { level = 'Needs-Work'; color = '#D4A017'; } else { level = 'Critical-Gap'; color = '#993C1D'; } grid.innerHTML += `
${domain}
${stats.correct} / ${stats.total} correct · ${pct}%
${level}
`; }); // Top 3 recommendations - rank weakest skills const allMissed = []; Object.entries(domains).forEach(([d, s]) => { Object.entries(s.missedSkills).forEach(([skill, count]) => { allMissed.push({ domain: d, skill, count }); }); }); allMissed.sort((a, b) => b.count - a.count); const top3 = allMissed.slice(0, 3); const recList = document.getElementById('recommendations-list'); recList.innerHTML = ''; if (top3.length === 0) { recList.innerHTML = '
Strong performance across all areas!
Maintain with periodic review and practice harder questions to push toward your target.
'; } else { top3.forEach((m, i) => { const impact = m.count >= 3 ? '+25-40 points' : (m.count >= 2 ? '+15-25 points' : '+5-15 points'); recList.innerHTML += `
${i+1}. ${m.skill} (${m.domain})
Missed ${m.count} question${m.count > 1 ? 's' : ''} in this skill. Estimated impact with focused practice: ${impact}
`; }); } showScreen('screen-report'); window.scrollTo(0, 0); }
